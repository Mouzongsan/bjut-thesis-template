% =============================================================
%  北京工业大学学位论文模板 (BJUT Thesis Template)

%           然后在各 chapt*.tex / cabstract.tex / eabstract.tex
%           等文件中填写内容，最后用 XeLaTeX 编译。
%  编译流程： xelatex main → bibtex main → xelatex main → xelatex main
% =============================================================

\documentclass{bjutthesis}

% ---- 常用宏包（按需保留 / 增删）----
\usepackage{slashed}
\usepackage{bm,mathrsfs}
\usepackage{empheq}
\usepackage{microtype}
\usepackage{longtable,array}

\allowdisplaybreaks[2]

% ---- cleveref 引用名（中文） ----
\crefformat{equation}{式~(#2#1#3)}
\Crefformat{equation}{式~(#2#1#3)}
\crefname{table}{表}{表}
\Crefname{table}{表}{表}
\crefname{figure}{图}{图}
\Crefname{figure}{图}{图}
\crefformat{chapter}{第#2#1#3章}
\Crefformat{chapter}{第#2#1#3章}
\crefname{section}{节}{节}
\Crefname{section}{节}{节}
\crefname{theorem}{定理}{定理}
\Crefname{theorem}{定理}{定理}
\crefname{lemma}{引理}{引理}
\Crefname{lemma}{引理}{引理}

% ---- 定理类环境（中文） ----
\newtheorem{theorem}{定理}[chapter]
\newtheorem{lemma}[theorem]{引理}
\newtheorem{proposition}[theorem]{命题}
\newtheorem{corollary}[theorem]{推论}
\newtheorem{definition}[theorem]{定义}
\newtheorem{remark}[theorem]{注}
\newenvironment{proof}{\par\noindent\textit{证明.}\ }{\hfill$\square$\par}

% ---- 自定义命令（示例，按需修改）----
\newcommand{\dd}{\mathop{}\!\mathrm{d}}
\newcommand{\Var}{\operatorname{Var}}
\newcommand{\Cov}{\operatorname{Cov}}
\def\cappendixname{附录}

% 中文圆括号公式编号
\makeatletter
\renewcommand{\tagform@}[1]{\maketag@@@{（#1）}}
\makeatother

% =============================================================
%  论文信息（请填写）
% =============================================================
\bjutset{
  clc            = {},                       % 中图分类号
  udc            = {},                       % UDC
  schoolcode     = {10005},                  % 学校代码（北工大 10005）
  secretlevel    = {公开},                    % 密级
  ctitle         = {论文中文题目},            % 中文题目
  cauthor        = {作者姓名},                % 作者姓名（中文）
  cdiscipline    = {学科名称},                % 学科
  csupervisor    = {导师姓名},                % 导师
  csupervstitle  = {},                       % 导师职称（可留空）
  cdate          = {YYYY年M月},              % 论文日期
  studentid      = {学号},                   % 学号
  etitle         = {English Title of the Thesis},
  cmajor         = {专业方向},
  cdegree        = {理学硕士},                % 学位（如：理学硕士 / 工学博士 等）
  ccollege       = {所在学院},
  corganization  = {北京工业大学},
}

\begin{document}

% ---- 图表标题中文设置 ----
\captionsetup[figure]{name=图}
\captionsetup[table]{name=表}
\renewcommand{\cftfigpresnum}{图~}
\renewcommand{\cfttabpresnum}{表~}

% ---- 封面与扉页 ----
\makecover
\cleardoublepage
\maketitle
\makestate

% ---- 前置部分：摘要、目录 ----
\frontmatter
\include{cabstract}     % 中文摘要
\include{eabstract}     % 英文摘要
\makebitoc              % 中英文双语目录

% ---- 主体部分：正文章节 ----
\mainmatter
% 章节标题使用中文（默认即为「第 X 章」格式，由 ctexbook 提供）
\ctexset{
  chapter = {
    name      = {第,章},
    number    = {\arabic{chapter}},
    aftername = \space,
  }
}
\include{chapt1}
\include{chapt2}
\include{chapt3}
\include{chapt4}
\include{chapt5}
\include{chapt6}

% ---- 参考文献与附录 ----
\include{references}
\include{appendices}
\cleardoublepage
\include{publication}   % 发表论文情况

% ---- 后置部分：致谢 ----
\backmatter
\include{acknowledgement}

\end{document}
